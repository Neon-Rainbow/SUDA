+ 后缀表达式求值
```C++
for i in string
{
    if i is number
    {
        stack.push(i);
    }
    else if i is operators
    {
        y = stack.pop;
        x = stack.pop;
        stack.push(x op y);
    }
}
return stack.top;
```
 + 中缀表达式转后缀
```C++
stack s1;
stack s2;
for i in string
    if i is number
        s2.push(i);
    if i is operators
        if s1 is empty or s1.top is "("
            s1.push(i);
        if i优先级>s1.top
            s1.push(i);
        else
            s2.push(s1.top)
    if i is bracket
        if i is left_bracket
            s1.push(i)
        if i is right_bracket
            s2.push(s1.pop) while s1.top is not right_bracket

```
