设计思路:

```C++
class Sort
{
public:
    Sort(int *a, int n);                       //构造函数
    ~Sort() = default;                         //析构函数
    void InsertionSort();                      //插入排序
    void ShellSort();                          //希尔排序
    void SelectSort();                         //选择排序
    void BubbleSort();                         //冒泡排序
    void QuickSort(int first, int last);       //快速排序
    void HeapSort();                           //堆排序
    void MergeSort(int p, int r);              //归并排序(递归)
    void MergeSort2();                         //归并排序(非递归)
    void Print();                              //输出数组
private:
    //快速排序的划分
    int Partition(int first, int last);
    //维护最大堆
    void MaxHeapify(int i);
    //构造最大堆
    void BuildMaxHeap();
	//堆的左节点
    int Left(int i)
    { return 2 * i + 1; }
	//堆的右节点
    int Right(int i)
    { return 2 * i; }
	//堆的父节点
    int Parent(int i)
    { return (i - 1) / 2; }
    //归并排序
    void Merge(int p, int q, int r);
    void MergePass(int h);

    int *data;     //int* data中存储数据
    int length;    //length存储数组的长度
```
